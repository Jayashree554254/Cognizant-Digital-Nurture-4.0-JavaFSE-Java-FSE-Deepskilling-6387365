// components/ComplaintRegister.js
import React, { useState } from 'react';

const ComplaintRegister = () => {
  const [employeeName, setEmployeeName] = useState('');
  const [complaint, setComplaint] = useState('');

  const handleSubmit = (e) => {
    e.preventDefault();

    if (!employeeName || !complaint) {
      alert('Please fill in both fields.');
      return;
    }

    // Generate a random reference number
    const referenceNumber = 'REF' + Math.floor(Math.random() * 1000000);

    alert(
      `Complaint submitted successfully!\nReference Number: ${referenceNumber}`
    );

    // Clear form
    setEmployeeName('');
    setComplaint('');
  };

  return (
    <div style={styles.container}>
      <h2>üõ†Ô∏è Complaint Register</h2>
      <form onSubmit={handleSubmit} style={styles.form}>
        <label>
          Employee Name:
          <input
            type="text"
            value={employeeName}
            onChange={(e) => setEmployeeName(e.target.value)}
            style={styles.input}
            required
          />
        </label>
        <br />
        <label>
          Complaint:
          <textarea
            value={complaint}
            onChange={(e) => setComplaint(e.target.value)}
            style={styles.textarea}
            required
          />
        </label>
        <br />
        <button type="submit" style={styles.button}>Submit Complaint</button>
      </form>
    </div>
  );
};

const styles = {
  container: {
    maxWidth: 500,
    margin: 'auto',
    padding: 20,
    border: '1px solid #ccc',
    borderRadius: 10,
    marginTop: 40,
    fontFamily: 'Arial',
  },
  form: {
    display: 'flex',
    flexDirection: 'column',
  },
  input: {
    width: '100%',
    padding: 8,
    marginTop: 5,
    marginBottom: 15,
  },
  textarea: {
    width: '100%',
    height: 100,
    padding: 8,
    marginTop: 5,
    marginBottom: 15,
  },
  button: {
    padding: 10,
    backgroundColor: '#007bff',
    color: '#fff',
    border: 'none',
    borderRadius: 4,
    cursor: 'pointer',
  },
};

export default ComplaintRegister;

}
